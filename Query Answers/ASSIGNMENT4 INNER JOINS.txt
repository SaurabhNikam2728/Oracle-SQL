1.SELECT ENAME,LOC FROM EMP,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO;
2.SELECT DNAME,SAL FROM DEPT,EMP WHERE DEPT.DEPTNO=EMP.DEPTNO;
3.SELECT SAL*12 ,DNAME FROM EMP,DEPT WHERE DEPT.DEPTNO=EMP.DEPTNO;
4.SELECT ENAME,DNAME FROM DEPT,EMP WHERE  DEPT.DEPTNO=EMP.DEPTNO AND DNAME LIKE '%A%';
5.SELECT DNAME,ENAME FROM DEPT,EMP WHERE DEPT.DEPTNO=EMP.DEPTNO AND JOB='SALESMAN';
6.SELECT DNAME,ENAME FROM EMP,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO AND JOB LIKE 'S%' AND DNAME LIKE 'S%';
7.SELECT DNAME,MGR FROM EMP,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO AND MGR<>7839;
8.SELECT DNAME,HIREDATE FROM EMP ,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO AND TO_CHAR(HIREDATE,'YY')>83 AND DNAME IN('ACCOUNTING','RESEARCH');
9.SELECT ENAME,DNAME FROM EMP,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO AND COMM IS NOT NULL AND EMP.DEPTNO IN(10,30);
10.SELECT DNAME,EMPNO  FROM EMP,DEPT WHERE EMP.DEPTNO=EMP.DEPTNO AND EMPNO IN(7839,7902) AND LOC='NEW YORK';
11.SELECT AVG(SAL) ,LOC FROM EMP,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO AND ENAME NOT LIKE '_A%' GROUP BY LOC;
12.SELECT ENAME ,LOC FROM EMP ,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO AND JOB='MANAGER' AND MGR IN (SELECT EMPNO FROM EMP WHERE EMPNO=7839);
13.SELECT DNAME ,EMPNO FROM EMP ,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO AND JOB='CLERK' AND MGR IS NOT NULL;
14.SELECT SUM(SAL) ,LOC FROM EMP,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO  GROUP BY LOC HAVING COUNT(*)>3;
15.SELECT DNAME,COUNT(*) FROM EMP,DEPT WHERE EMP.DEPTNO=DEPT.DEPTNO GROUP BY EMP.DEPTNO ,DNAME;
